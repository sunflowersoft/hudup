SELECT distinct(`SAM_ASSESSMENTGRADING_T`.`AGENTID`)
FROM `SAM_ASSESSMENTGRADING_T`, `SAM_ITEMGRADING_T`, `GB_GRADABLE_OBJECT_T`, `GB_CATEGORY_T`
WHERE `SAM_ASSESSMENTGRADING_T`.`ASSESSMENTGRADINGID` = `SAM_ITEMGRADING_T`.`ASSESSMENTGRADINGID`
AND `SAM_ASSESSMENTGRADING_T`.`PUBLISHEDASSESSMENTID` = `GB_GRADABLE_OBJECT_T`.`EXTERNAL_ID` 
AND `GB_GRADABLE_OBJECT_T`.`CATEGORY_ID` = `GB_CATEGORY_T`.`ID`
AND `GB_CATEGORY_T`.`OPTION_CATE` = 1
AND `GB_GRADABLE_OBJECT_T`.`REMOVED` = 0


SELECT distinct(`SAM_ITEMGRADING_T`.`PUBLISHEDITEMID`)
FROM `SAM_ASSESSMENTGRADING_T`, `SAM_ITEMGRADING_T`, `GB_GRADABLE_OBJECT_T`, `GB_CATEGORY_T`
WHERE `SAM_ASSESSMENTGRADING_T`.`ASSESSMENTGRADINGID` = `SAM_ITEMGRADING_T`.`ASSESSMENTGRADINGID`
AND `SAM_ASSESSMENTGRADING_T`.`PUBLISHEDASSESSMENTID` = `GB_GRADABLE_OBJECT_T`.`EXTERNAL_ID` 
AND `GB_GRADABLE_OBJECT_T`.`CATEGORY_ID` = `GB_CATEGORY_T`.`ID`
AND `GB_CATEGORY_T`.`OPTION_CATE` = 1
AND `GB_GRADABLE_OBJECT_T`.`REMOVED` = 0


SELECT `SAM_ASSESSMENTGRADING_T`.`AGENTID`, `SAM_ITEMGRADING_T`.`PUBLISHEDITEMID`, `SAM_ASSESSMENTGRADING_T`.`PUBLISHEDASSESSMENTID`, `SAM_ITEMGRADING_T`.`AUTOSCORE`
FROM `SAM_ASSESSMENTGRADING_T`, `SAM_ITEMGRADING_T`, `GB_GRADABLE_OBJECT_T`, `GB_CATEGORY_T`
WHERE `SAM_ASSESSMENTGRADING_T`.`ASSESSMENTGRADINGID` = `SAM_ITEMGRADING_T`.`ASSESSMENTGRADINGID`
AND `SAM_ASSESSMENTGRADING_T`.`PUBLISHEDASSESSMENTID` = `GB_GRADABLE_OBJECT_T`.`EXTERNAL_ID` 
AND `GB_GRADABLE_OBJECT_T`.`CATEGORY_ID` = `GB_CATEGORY_T`.`ID`
AND `GB_CATEGORY_T`.`OPTION_CATE` = 1
AND `GB_GRADABLE_OBJECT_T`.`REMOVED` = 0
AND `SAM_ITEMGRADING_T`.`AUTOSCORE` IS NOT NULL
GROUP BY `SAM_ASSESSMENTGRADING_T`.`AGENTID`, `SAM_ITEMGRADING_T`.`PUBLISHEDITEMID`
